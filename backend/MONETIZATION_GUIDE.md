## Payment Integration

### Paystack Setup (Nigerian Payment Gateway):

```javascript
// backend/src/routes/subscriptions.js
const paystack = require('paystack')(process.env.PAYSTACK_SECRET_KEY);

router.post('/subscribe', protect, async (req, res) => {
  const { plan } = req.body; // 'monthly' or 'yearly'
  
  const amount = plan === 'monthly' ? 2000 : 20000;
  
  const response = await paystack.transaction.initialize({
    email: req.user.email,
    amount: amount * 100, // Kobo
    metadata: {
      userId: req.user._id,
      plan
    },
    callback_url: `${process.env.API_URL}/api/subscriptions/verify`
  });
  
  res.json({
    success: true,
    data: {
      authorizationUrl: response.data.authorization_url,
      reference: response.data.reference
    }
  });
});

router.post('/verify', async (req, res) => {
  const { reference } = req.body;
  
  const response = await paystack.transaction.verify(reference);
  
  if (response.data.status === 'success') {
    // Update user subscription
    const user = await User.findById(response.data.metadata.userId);
    user.isPremium = true;
    user.premiumExpiry = new Date(Date.now() + 30 * 24 * 60 * 60 * 1000);
    await user.save();
    
    res.json({ success: true, message: 'Subscription activated' });
  }
});
```
